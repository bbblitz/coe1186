  Curved track
    cur, x, y, start_angle, angle_length, radius, block_clockwise, block_counterclockwise, block_clockwise_to, block_counterclockwise_to, block_id, trackcontroller_id

      cur 20, 20, 0, 90, 100, 4, 8, HEAD, TAIL, 2, 0
